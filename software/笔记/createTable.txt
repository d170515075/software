CREATE TABLE `oils_stock` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT, 
   `weight`decimal(18,2),
  `price` decimal(18,2), 
  `name` varchar(50),
  `enter_sell`varchar(100),
  `gun` varchar(100),
  `varieties_surplus` decimal(18,2),
  `total_surplus` decimal(18,2),
  `time` datetime,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
